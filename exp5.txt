import 'package:flutter/material.dart';

void main() => runApp(MyApp());

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'College App',
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: MyHomePage(),
    );
  }
}

class MyHomePage extends StatefulWidget {
  @override
  _MyHomePageState createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  // For tracking the selected menu
  int _selectedIndex = 0;

  // Method to update the selected menu
  void _onMenuItemSelected(int index) {
    setState(() {
      _selectedIndex = index;
    });
    Navigator.of(context).pop(); // Close the drawer
  }

  // Pages corresponding to the menu items
  final List<Widget> _pages = [
    AboutUsPage(),
    DepartmentsPage(),
    StudentSectionPage(),
    ContactUsPage(),
  ];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('College App'),
      ),
      // Drawer widget
      drawer: Drawer(
        child: ListView(
          padding: EdgeInsets.zero,
          children: <Widget>[
            DrawerHeader(
              decoration: BoxDecoration(
                color: Colors.blue,
              ),
              child: Text(
                'College Name',
                style: TextStyle(
                  color: Colors.white,
                  fontSize: 24,
                ),
              ),
            ),
            ListTile(
              title: Text('About Us'),
              onTap: () => _onMenuItemSelected(0),
            ),
            ListTile(
              title: Text('Departments'),
              onTap: () => _onMenuItemSelected(1),
            ),
            ListTile(
              title: Text('Student Section'),
              onTap: () => _onMenuItemSelected(2),
            ),
            ListTile(
              title: Text('Contact Us'),
              onTap: () => _onMenuItemSelected(3),
            ),
          ],
        ),
      ),
      body: _pages[_selectedIndex], // Show the corresponding page
    );
  }
}

class AboutUsPage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Center(
      child: Text('About Us', style: TextStyle(fontSize: 24)),
    );
  }
}

class DepartmentsPage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Center(
      child: Text('Departments', style: TextStyle(fontSize: 24)),
    );
  }
}

class StudentSectionPage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Center(
      child: Text('Student Section', style: TextStyle(fontSize: 24)),
    );
  }
}

class ContactUsPage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Center(
      child: Text('Contact Us', style: TextStyle(fontSize: 24)),
    );
  }
}
